A message containing letters from A-Z is being encoded to numbers using the following mapping:
'A' -> 1
'B' -> 2
...
'Z' -> 26
Given an encoded message containing digits, determine the total number of ways to decode it.
For example,â€¨Given encoded message "12", it could be decoded as "AB" (1 2) or "L" (12).
The number of ways decoding "12" is 2.

class Solution {
public:
    int numDecodings(string s) {
        int len = s.length();
        if(len==0) return 0;
        if(s[0] == '0'){
            return 0;
        }
        if(len == 1) return 1;

        vector<int> v(len+1, 0);
        v[0] = 1;
        v[1] = 1;
        for(int i=2; i<=len;i++){
            if(s[i-1]>='1' && s[i-1] <='6'){
                if(s[i-2] =='1' || s[i-2] == '2' ){
                    v[i] = v[i-1]+v[i-2];
                }
                else{
                    v[i] = v[i-1];
                }
            }
            else if(s[i-1] == '0'){
                if(s[i-2] == '1' || s[i-2] == '2'){
                    v[i] = v[i-2];
                }
                else{
                    v[i] = 0;
                }
            }
            else if(s[i-1] >= '7' && s[i-1] <= '9'){
                if(s[i-2] == '1'){
                    v[i] = v[i-1]+v[i-2];
                }
                else{
                    v[i] = v[i-1];
                }
            }
        }
        return v[len];
    }
};
